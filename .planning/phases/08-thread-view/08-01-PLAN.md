```go
// RelativeDate formats a time relative to now for thread summary display.
func RelativeDate(t time.Time) string {
    now := time.Now()
    diff := now.Sub(t)
    switch {
    case diff < 24*time.Hour:
        return t.Format("3:04 PM")
    case diff < 7*24*time.Hour:
        return t.Format("Mon 3:04 PM")
    case t.Year() == now.Year():
        return t.Format("Jan 2")
    default:
        return t.Format("Jan 2, 2006")
    }
}

// Truncate truncates a string to n runes, appending ellipsis if truncated.
func Truncate(s string, n int) string {
    runes := []rune(s)
    if len(runes) <= n {
        return s
    }
    return string(runes[:n]) + "..."
}
```

**ThreadMessageCard component** — `templ ThreadMessageCard(msg query.MessageSummary, isLatest bool, highlightID int64)`:

```html
<details
    id="msg-{msg.ID}"
    class="thread-message {thread-focused if msg.ID == highlightID}"
    data-msg-id="{msg.ID}"
    open if isLatest || msg.ID == highlightID
>
    <summary class="thread-message-summary">
        <span class="thread-msg-from">{displayName}</span>
        <span class="thread-msg-date">{RelativeDate(msg.SentAt)}</span>
        <span class="thread-msg-snippet">{Truncate(msg.Snippet, 80)}</span>
    </summary>

    if isLatest || msg.ID == highlightID {
        <!-- Eagerly loaded body -->
        <div class="thread-message-expanded">
            <dl class="header-list">
                <dt>From</dt><dd>{displayName} &lt;{msg.FromEmail}&gt;</dd>
                <dt>Date</dt><dd>{FormatTime(msg.SentAt)}</dd>
                <dt>Size</dt><dd>{FormatBytes(msg.SizeEstimate)}</dd>
                if len(msg.Labels) > 0 {
                    <dt>Labels</dt><dd><span class="label-list">...labels...</span></dd>
                }
            </dl>
            <div id="email-body-wrapper-{msg.ID}" class="email-render-wrapper">
                <div class="email-images-banner">
                    <span>External images blocked.</span>
                    <a href="#"
                       hx-get="/messages/{msg.ID}/body-wrapper?showImages=true"
                       hx-target="closest .email-render-wrapper"
                       hx-swap="outerHTML">Load images</a>
                </div>
                <iframe
                    src="/messages/{msg.ID}/body"
                    sandbox="allow-scripts allow-popups allow-popups-to-escape-sandbox"
                    class="email-iframe"
                    scrolling="no"
                    frameborder="0"
                ></iframe>
            </div>
            if msg.HasAttachments {
                <div class="thread-attachments">
                    <span class="attachment-count">{msg.AttachmentCount} attachment(s)</span>
                    <a href="/messages/{msg.ID}" class="attachment-detail-link">View details</a>
                </div>
            }
        </div>
    } else {
        <!-- Lazy-load placeholder -->
        <div
            class="thread-message-body"
            hx-get="/messages/{msg.ID}/body-wrapper"
            hx-trigger="toggle once"
            hx-swap="innerHTML"
        >
            <p class="loading-placeholder">Loading message...</p>
        </div>
    }
</details>
```

**CRITICAL — ID collision handling for "Load images":**
For the eagerly loaded messages, use `hx-target="closest .email-render-wrapper"` instead of `hx-target="#email-body-wrapper"` on the "Load images" link. This scopes the HTMX swap to the nearest wrapper ancestor, avoiding the multi-ID collision issue identified in research. The `messageBodyWrapper` endpoint response still uses `id="email-body-wrapper"` but since `closest` targets the DOM-nearest wrapper, this is safe.

For lazy-loaded messages (the `hx-get="/messages/{id}/body-wrapper"` placeholder), use `hx-swap="innerHTML"` — the response from `messageBodyWrapper` is the wrapper div which becomes the inner content of `.thread-message-body`. The response's "Load images" button targets `#email-body-wrapper` which will be the only one of that ID within this message's context (the lazy load replaces innerHTML, so only one wrapper per expanded message exists at a time).
