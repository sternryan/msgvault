```go
func MaxAggregateCount(rows []query.AggregateRow) int64 {
    var max int64
    for _, r := range rows {
        if r.Count > max {
            max = r.Count
        }
    }
    return max
}
```

6. Add `BarPercent` function:
```go
func BarPercent(count, maxCount int64) string {
    if maxCount == 0 {
        return "0"
    }
    pct := float64(count) / float64(maxCount) * 100
    return fmt.Sprintf("%.1f", pct)
}
```
